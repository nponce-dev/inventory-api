${p.name} ${p.sku} · $${p.precio_venta != null ? p.precio_venta.toFixed(2) : '—'}
`).join(''); } function selectDiscProduct(id, name, precio) { document.getElementById('disc-product-id').value = id; document.getElementById('disc-product-name').value = name; document.getElementById('disc-product-precio').value = precio; document.getElementById('disc-search').value = name; document.getElementById('disc-suggestions').style.display = 'none'; } function addDiscItem() { const product_id = parseInt(document.getElementById('disc-product-id').value); const name = document.getElementById('disc-product-name').value; const precio = parseFloat(document.getElementById('disc-product-precio').value) || 0; const quantity = parseInt(document.getElementById('disc-qty').value); if (!product_id) { toast('Seleccioná un producto de la lista', 'error'); return; } if (isNaN(quantity) || quantity < 1) { toast('Ingresá una cantidad válida', 'error'); return; } const existing = discItems.find(i => i.product_id === product_id); if (existing) { existing.quantity += quantity; } else { discItems.push({ product_id, name, quantity, precio }); } renderDiscItems(); document.getElementById('disc-search').value = ''; document.getElementById('disc-product-id').value = ''; document.getElementById('disc-product-name').value = ''; document.getElementById('disc-product-precio').value = ''; document.getElementById('disc-qty').value = ''; } function renderDiscItems() { const list = document.getElementById('disc-items-list'); const counter = document.getElementById('disc-items-count'); const sumBox = document.getElementById('disc-precio-normal'); const sumEl = document.getElementById('disc-sum'); if (!discItems.length) { list.style.display = 'none'; sumBox.style.display = 'none'; counter.textContent = 'Ningún producto agregado todavía'; return; } const sum = discItems.reduce((acc, i) => acc + (i.precio * i.quantity), 0); list.style.display = 'flex'; sumBox.style.display = 'block'; sumEl.textContent = `$${sum.toFixed(2)}`; counter.textContent = `${discItems.length} producto(s) en este combo`; list.innerHTML = discItems.map((item, idx) => `
${item.name} ${item.quantity} u. · $${(item.precio * item.quantity).toFixed(2)}
`).join(''); } function removeDiscItem(idx) { discItems.splice(idx, 1); renderDiscItems(); } async function saveDiscount() { const name = document.getElementById('disc-name').value.trim(); const precio_descuento = parseFloat(document.getElementById('disc-precio').value); if (!name) { toast('Ingresá un nombre para el descuento', 'error'); return; } if (!discItems.length) { toast('Agregá al menos un producto al combo', 'error'); return; } if (isNaN(precio_descuento) || precio_descuento <= 0) { toast('Ingresá el precio del combo', 'error'); return; } const body = { name, precio_descuento, items: discItems.map(i => ({ product_id: i.product_id, quantity: i.quantity })) }; try { const res = await fetch(API + '/discounts/', { method: 'POST', headers: authHeaders(), body: JSON.stringify(body) }); if (res.ok) { toast('Descuento guardado correctamente'); discItems = []; renderDiscItems(); document.getElementById('disc-name').value = ''; document.getElementById('disc-precio').value = ''; loadDiscounts(); } else { const err = await res.json(); toast(err.detail || 'Error al guardar', 'error'); } } catch { toast('No se pudo conectar con la API', 'error'); } } async function loadDiscounts() { try { const [discRes, productsRes] = await Promise.all([ fetch(API + '/discounts/', { headers: authHeaders() }), fetch(API + '/products/', { headers: authHeaders() }) ]); const discounts = await discRes.json(); const products = await productsRes.json(); const productMap = {}; products.forEach(p => productMap[p.id] = { name: p.name, precio: p.precio_venta }); const tbody = document.getElementById('discounts-tbody'); if (!discounts.length) { tbody.innerHTML = 'No hay descuentos guardados todavía'; return; } tbody.innerHTML = discounts.map(d => { const itemsText = d.items.map(i => { const p = productMap[i.product_id]; return p ? `${p.name} ×${i.quantity}` : `#${i.product_id} ×${i.quantity}`; }).join(', '); const precioNormal = d.items.reduce((acc, i) => { const p = productMap[i.product_id]; return acc + (p && p.precio ? p.precio * i.quantity : 0); }, 0); const ahorro = precioNormal > 0 ? precioNormal - d.precio_descuento : null; const ahorroText = ahorro !== null && ahorro > 0 ? `-$${ahorro.toFixed(2)}` : `—`; const normalText = precioNormal > 0 ? `$${precioNormal.toFixed(2)}` : '—'; return ` #${d.id} ${d.name} ${itemsText} ${normalText} $${d.precio_descuento.toFixed(2)} ${ahorroText} `; }).join(''); } catch { document.getElementById('discounts-tbody').innerHTML = 'No se pudo conectar con la API'; } } async function deleteDiscount(id) { if (!confirm('¿Eliminar este descuento?')) return; try { const res = await fetch(API + '/discounts/' + id, { method: 'DELETE', headers: authHeaders() }); if (res.ok) { toast('Descuento eliminado'); loadDiscounts(); } else { toast('Error al eliminar', 'error'); } } catch { toast('No se pudo conectar con la API', 'error'); } }
